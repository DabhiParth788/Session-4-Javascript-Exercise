function task1(string) {
  let sum = 0;
  let tempNumber = "";
  for (let char of string) {
    if (!isNaN(char) && char !== " ") {
      tempNumber += char;
    } else if (tempNumber.length > 0) {
      sum += parseInt(tempNumber);
      tempNumber = "";
    }
  }
  if (tempNumber.length > 0) {
    sum += parseInt(tempNumber);
  }
  return sum;
}
